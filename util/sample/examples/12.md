const flip = x=>({ heads:random_boolean(ph) })

const wake = ft => at_(ft, inc(x=>!x.awake, x=>{ x.awake=true }))
const events = [
  do_(inc('days')),    // increment x.days (at midnight)
  do_(inc(flip)),      // flip a coin (at midnight)
  sleep(hour(3)),      // sleep at 3am
  wake(hours(9,10)),   // wake up at 9-10am
]
const x3 = advance(x0, events, 3)
confine(x3.heads, 2)
