// defining event definition helpers here is more efficient
// also significantly improves readability of simulation code above
const awake = x => x.awake
const asleep = x => !x.awake
const awake_and = (fc) => (x) => fc(x) && awake(x)
const sleep = ft => _if(awake, ft, x=>x.awake=false)
const wake =  ft => _if(asleep, ft, x=>(x.awake=true,x._breakfast=0))
const breakfast = ft => _if(awake_and(x=>!x._breakfast), ft, inc('_breakfast', .5, 'breakfasts'))
const flip = ph => _do(inc(x=>({ flips:1, heads:random_boolean(ph)?1:0, _:{ph} })))

const _sample_options = { 
  stats:'mks ess wsum r t',
  max_time: 5000,
  plot:true,
  size:1,
}
