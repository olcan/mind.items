const reprice = _do(x=>{
  apply(x.prices, (x,a)=> x * exp(dA[a] + σA[a]*random_normal()))
  x.value = sum(A, a=> x.alloc[a] * x.prices[a])
})
const events = name_events(()=>reprice)
