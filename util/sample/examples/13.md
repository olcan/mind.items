const A = 4
const pA = sample_array(A, summand(1), 
  {names: ['crypto', 'stocks', 'bonds', 'reits']})
const μA = [.05, .025, .005, .0125]
const σA = [.1, .05, .01, .025]
const x0 = { t: 0, prices:array(A,1), value:1 }
const xt = simulate(x0, 10, _do(x=>{
  apply(x.prices, (x,a)=> x * exp(μA[a]-.5*σA[a]**2 + σA[a]*random_normal()))
  x.value = sum(A, a=> pA[a] * x.prices[a])
}))
confine(xt.value, or(below(1), above(2)))
plot(xt.value)
