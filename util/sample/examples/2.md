sample(() => {
  let a = sample(uniform(0, 1)) // uniform prior P(A) on [0,1]
  let b = sample(uniform(a, 1)) // uniform prior P(B|a) on [a,1]

}, {stats:'tks ess elw', plot:true, targets });
