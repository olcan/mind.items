// TODO: define a general temporal process using #util/events
//       sample inputs/params, confine states (intermediate or final)

const ph = sample(between(0,1))
const h0 = sample(index(5))
const x0 = { t:0, days:0, heads:h0, awake:true, _events:[], _states:[] }
const eJ = [
  // increment days at midnight
  do_(x=>x.days++),
  // flip a coin at midnight
  do_(param((x,θ)=>x.heads+=θ.h?1:0, x=>({h:random_boolean(ph)}))),
  // sleep at 3am when awake
  do_(x=>x.awake=false, hour(3), x=>x.awake),
  // wake up at 9-10am when asleep
  do_(x=>x.awake=true, hours(9,10), x=>!x.awake),
]
const x3 = advance(x0, eJ, 3)
confine(x3.heads, 2)
