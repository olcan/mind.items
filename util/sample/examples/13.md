const A = 4 // assets
const T = 10 // time periods
const μA = [.05, .025, .005, .0125] // mean change (relative) per period
const σA = [.1, .05, .01, .025] // standard deviation in change per period
const dA = array(A, a=> μA[a]-.5*σA[a]**2)
const pA = array(A) // asset allocation
const update_prices = _do(x=>{
  apply(x.prices, (x,a)=> x * exp(dA[a] + σA[a]*random_normal()))
  x.value = sum(A, a=> pA[a] * x.prices[a])
})
update_prices._name = 'update_prices'

const _sample_options = { 
  params:{A},
  stats:'mks ess wsum r t',
  max_time: 10000,
  table:true,
  plot:true,
  // log:true,
  // size:1, // debug
}
