const mean_sleep_length = mean_wake_hour - mean_sleep_hour
const user_days = (Date.now() - _user.oldest_item_time) * _1ms
const user_hours_awake = user_days * (24 - mean_sleep_length)
const awake_hours_per_create = user_hours_awake / _user.items

const wake = ft => _if(x=>!x.awake, ft, x=>{ x.awake=true })
const sleep = ft => _if(x=>x.awake, ft, x=>{ x.awake=false })
const create_item = ft => _if(x=>x.awake, ft, inc({ items:1 }))
const delete_item = ft => _if(x=>x.awake, ft, inc({ items:-1 }))

const events = name_events(
  () => wake(daily(within(mean_wake_hour, 1))),
  () => sleep(daily(within(mean_sleep_hour, 1))),
  () => create_item(randomly(awake_hours_per_create)),
  () => delete_item(randomly(awake_hours_per_delete)),
)
